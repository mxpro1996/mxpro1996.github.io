中断接收

void USART3_IRQHandler(void)                	//串口3中断服务程序
{
    u8 Res;
    if(__HAL_UART_GET_IT_SOURCE(&huart3, UART_IT_RXNE) != RESET)  //接收中断
    {
        Res = USART3->DR;	//读取接收到的数据
        USART3_RX_BUF[USART3_RX_STA]=Res;
     
//		USART1->DR = Res;
//		while(__HAL_UART_GET_FLAG(&huart1,UART_FLAG_TXE)!=SET){}

		
        if(USART3_RX_BUF[USART3_RX_STA] == 'K')  // 表示接收到了“OK"回应
        {
            if(USART3_RX_BUF[USART3_RX_STA - 1] == 'O')
            {
                Flag_USART3_receive_OK = 1;  //OK接收成功
            }
        }
        USART3_RX_STA++;
        if(USART3_RX_STA > (USART3_REC_LEN - 1))
            USART3_RX_STA = 0;
    }
}
